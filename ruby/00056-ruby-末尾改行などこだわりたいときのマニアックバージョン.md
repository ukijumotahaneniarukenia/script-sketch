ruby -e 'p [""]*"@"'

ruby -e 'p []*"@"'
""


ruby -e 'p ["a"]*"@"'
"a"

ruby -e 'p ["a","b"]*"@"'
"a@b"


ruby -e 'p ["a","b","c"]*"@"'
"a@b@c"

ruby -e 'p +?\n'
"\n"

ruby -e 'p ""+?\n'
"\n"

ruby -e 'p "a"+?\n'
"a"

ruby -e 'p "ab"+?\n'
"ab\n"

"a@b@c"

ruby -e 'p ["a","b","c"]*?\n'
"a\nb\nc"
