post = "さようなら"

pre = pre.gsub(/#{pattern}/,post)
p pre

pre = pre.gsub(/#{pattern}/,post)
p pre

pattern = "\\p{Hiragana}+"
pre = pre.gsub(/#{pattern}/,post)
p pre

pattern = "(?:\\p{Hiragana}|\\p{Katakana}|[ー－]|[一-龠々])+" #全角ひらがな、全角カタカナ、漢字（鬼車）
pre = pre.gsub(/#{pattern}/,post)
p pre
