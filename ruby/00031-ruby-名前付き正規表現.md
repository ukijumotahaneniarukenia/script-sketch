ruby -e 'm = /(?<year>\d+)-(?<month>\d+)-(?<day>\d+)/.match("2016-09-01");p m'
#<MatchData "2016-09-01" year:"2016" month:"09" day:"01">


ruby -e 'm = /(?<year>\d+)-(?<month>\d+)-(?<day>\d+)/.match("2016-09-01");p m.values_at(1,2,3)'
["2016", "09", "01"]


ruby -e 'm = /(?<year>\d+)-(?<month>\d+)-(?<day>\d+)/.match("2016-09-01");p m.values_at(1..3)'
["2016", "09", "01"]

ruby -e 'm = /(?<year>\d+)-(?<month>\d+)-(?<day>\d+)/.match("2016-09-01");p m.values_at(1..m.size)'
["2016", "09", "01", nil]

ruby -e 'm = /(?<year>\d+)-(?<month>\d+)-(?<day>\d+)/.match("2016-09-01");p m.values_at(1..(m.size-1))'
["2016", "09", "01"]


ruby -e 'm = /(?<year>\d+)-(?<month>\d+)-(?<day>\d+)/.match("2016-09-01");p m.values_at("year")'
["2016"]


ruby -e 'm = /(?<year>\d+)-(?<month>\d+)-(?<day>\d+)/.match("2016-09-01");p m.values_at("year","month")'
["2016", "09"]


ruby -e 'm = /(?<year>\d+)-(?<month>\d+)-(?<day>\d+)/.match("2016-09-01");p m.values_at("year","month","day")'
["2016", "09", "01"]


ruby -e 'm = /(?<year>\d+)-(?<month>\d+)-(?<day>\d+)/.match("2016-09-01");p m.named_captures'
{"year"=>"2016", "month"=>"09", "day"=>"01"}
