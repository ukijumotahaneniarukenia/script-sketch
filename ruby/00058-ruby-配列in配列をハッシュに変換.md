pythonのタプルのリストをzipにすると辞書型になるやつと一緒

ppはプリティプリント

```
ruby -e'p [*12354..12367].sample(10).map(&->{[_1,_1.chr("utf-8")]})' 
[[12366, "ぎ"], [12358, "う"], [12356, "い"], [12355, "ぃ"], [12362, "お"], [12359, "ぇ"], [12363, "か"], [12354, "あ"], [12367, "く"], [12357, "ぅ"]]


ruby -e'p [*12354..12367].sample(10).map(&->{[_1,_1.chr("utf-8")]}).to_h' 
{12359=>"ぇ", 12356=>"い", 12367=>"く", 12355=>"ぃ", 12357=>"ぅ", 12363=>"か", 12354=>"あ", 12360=>"え", 12365=>"き", 12364=>"が"}


ruby -e'pp [*12354..12367].sample(10).map(&->{[_1,_1.chr("utf-8")]})' 
[[12356, "い"],
 [12355, "ぃ"],
 [12354, "あ"],
 [12362, "お"],
 [12363, "か"],
 [12360, "え"],
 [12365, "き"],
 [12357, "ぅ"],
 [12367, "く"],
 [12361, "ぉ"]]
 
 ruby -e'pp [*12354..12367].sample(10).map(&->{[_1,_1.chr("utf-8")]}).to_h' 
{12358=>"う",
 12363=>"か",
 12357=>"ぅ",
 12362=>"お",
 12364=>"が",
 12366=>"ぎ",
 12367=>"く",
 12361=>"ぉ",
 12360=>"え",
 12355=>"ぃ"}
```
