- IN

```
$ jq -n '{"a":[{"b":1}]}'
{
  "a": [
    {
      "b": 1
    }
  ]
}
```

- CMD

```
$ jq -nc '{"a":[{"b":1}]}|path(..)'
```

- OUT

```
[]
["a"]
["a",0]
["a",0,"b"]
```

- CMD

```
$ jq -nc '{"a":[{"b":1}]}|getpath([])'
```

- OUT

```
{"a":[{"b":1}]}
```

- CMD

```
jq -nc '{"a":[{"b":1}]}|getpath(["a",0,"b"])'
```


- OUT

```
1
```

- CMD

```
jq -nc '{"a":[{"b":1}]}|getpath(["a",0])'
```


- OUT

```
{"b":1}
```
