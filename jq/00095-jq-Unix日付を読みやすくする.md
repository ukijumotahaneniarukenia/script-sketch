IN

```
$ echo '{"time":[1500000000,1500100000,1500200000,1500300000]}' | jq
{
  "time": [
    1500000000,
    1500100000,
    1500200000,
    1500300000
  ]
}


$ echo '{"time":[1500000000,1500100000,1500200000,1500300000]}' | jq '.time[]'
1500000000
1500100000
1500200000
1500300000
```

CMD

```
$ echo '{"time":[1500000000,1500100000,1500200000,1500300000]}' | jq '.time[]|=strftime("%Y-%m-%dT%H:%M:%S")'
```

OUT

```
{
  "time": [
    "2017-07-14T02:40:00",
    "2017-07-15T06:26:40",
    "2017-07-16T10:13:20",
    "2017-07-17T14:00:00"
  ]
}
```

文字列の場合は数値型に変換してから実行

IN

```
$ echo '{"time":["1500000000","1500100000","1500200000","1500300000"]}' | jq
{
  "time": [
    "1500000000",
    "1500100000",
    "1500200000",
    "1500300000"
  ]
}
```

CMD

```
$ echo '{"time":["1500000000","1500100000","1500200000","1500300000"]}' | jq '.time[] |= (tonumber|strftime("%Y-%m-%dT%H:%M:%S"))'
```

OUT

```
{
  "time": [
    "2017-07-14T02:40:00",
    "2017-07-15T06:26:40",
    "2017-07-16T10:13:20",
    "2017-07-17T14:00:00"
  ]
}
```

IN

```
$ echo '{"time":[{"start":1500000000,"end":1500100000},{"start":1500200000,"end":1500300000}]}' | jq
{
  "time": [
    {
      "start": 1500000000,
      "end": 1500100000
    },
    {
      "start": 1500200000,
      "end": 1500300000
    }
  ]
}

$ echo '{"time":[{"start":1500000000,"end":1500100000},{"start":1500200000,"end":1500300000}]}' | jq '.time[]'
{
  "start": 1500000000,
  "end": 1500100000
}
{
  "start": 1500200000,
  "end": 1500300000
}
```

CMD

```
$ echo '{"time":[{"start":1500000000,"end":1500100000},{"start":1500200000,"end":1500300000}]}' | jq '.time[]|=(map(.|strftime("%Y-%m-%dT%H:%M:%S")))'
```

OUT

```
{
  "time": [
    [
      "2017-07-14T02:40:00",
      "2017-07-15T06:26:40"
    ],
    [
      "2017-07-16T10:13:20",
      "2017-07-17T14:00:00"
    ]
  ]
}
```
