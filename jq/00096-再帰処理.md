CMD

```
$ echo 'abcdef' | jq -Rr 'split("^..\\K";"g") as [$f,$r]|$r,$f'
```

OUT

```
cdef
ab
```

CMD

```
$ echo 'abcdefghijk' | jq -Rr 'def UNKO(first;remain):first,(remain|select(.!=null)|split("^...\\K";"g")) as [$f,$r]|UNKO($f;$r);split("^...\\K";"g") as [$f,$r]|UNKO($f;$r)|select(.!="")'
```

OUT

```
abc
def
ghi
jk
```

CMD

```
$ echo abcdefghi | jq -Rr --arg reg_exp "^...\\K" 'split($reg_exp;"g")'
[
  "abc",
  "defghi"
]
```

```
$ echo 'abcdefghijk' | jq -Rr --arg reg_exp "^...\\K" 'def UNKO(first;remain):first,(remain|select(.!=null)|split($reg_exp;"g")) as [$f,$r]|UNKO($f;$r);split($reg_exp;"g") as [$f,$r]|UNKO($f;$r)|select(.!="")'
abc
def
ghi
jk
```
