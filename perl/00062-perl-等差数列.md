```
$perl -E '$,="\t";my($n,@m)=@ARGV;my($i,%rt)=();map{++$i;if(1==$i){$rt{$i}=$n;say $rt{$i},"$rt{$i}"}else{$rt{$i}=$rt{$i-1}+$n*($i-($i-1));say $rt{$i},"$rt{$i-1}+$n*($i-($i-1))"}}@m' 3 {1..9}
```
