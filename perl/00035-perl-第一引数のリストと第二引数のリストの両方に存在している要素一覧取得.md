```
$perl -e 'my ($self,$other)=@ARGV;my @self=`$self`;my @other=`$other`;my %p=();map{--$p{$_}}@self;print grep{++$p{$_}==1} @other' 'ls 0000[0-3]*'  'ls 000[01][0-3]*'
```
