- LIB

```
$sudo Rscript -e 'install.packages("lubridate", repos = "http://cran.ism.ac.jp/",type = "source", dependencies = TRUE, Ncpus = getOption("Ncpus", 12L))'
```


- CMD

```
$Rscript -e 'library(lubridate);ymd("2017-1-19")' 2>/dev/null
[1] "2017-01-19"
$Rscript -e 'library(lubridate);class(ymd("2017-1-19"));' 2>/dev/null
[1] "Date"
```

ラッパーの模様

```
$Rscript -e 'library(lubridate);ymd_hms("2019-01-19 08:30:10");' 2>/dev/null
[1] "2019-01-19 08:30:10 UTC"
$Rscript -e 'library(lubridate);class(ymd_hms("2019-01-19 08:30:10"));' 2>/dev/null
[1] "POSIXct" "POSIXt"
```


```
$Rscript -e 'library(lubridate);today();' 2>/dev/null
[1] "2020-03-22"
$Rscript -e 'library(lubridate);class(today());' 2>/dev/null
[1] "Date"
$Rscript -e 'library(lubridate);now();' 2>/dev/null
[1] "2020-03-22 12:16:41 JST"
$Rscript -e 'library(lubridate);class(now());' 2>/dev/null
[1] "POSIXct" "POSIXt"
$
```


- CMD
  - 日付のベクタ生成
```
$Rscript -e 'library(lubridate);c<-seq(0,30,1);unlist(lapply(c,function(e){return (as.character(e+today()))}));' 2>/dev/null
```
